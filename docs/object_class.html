 93    def check_constraints(self, next_act):
 94        """
 95        Checks whether the next transition to be executed has any relationships with transitions of
 96        other objects. If such relationships exist, the method extracts the corresponding parameters
 97        from the input.json file.
 98        For instance:
 99        ```json
100        "Name of the transition": {
101            "obj": "item",
102            "trans": ["Pick Item", "Remove Item"],
103            "card": "All | Any | int | [min, max] | CUSTOM"
104        }
105        ```
106        """
107
108        info = self._object_params["object_constraints"][next_act]
109
110        obj_type = info.get("obj")
111        trans_list = info.get("trans", [])
112        cardinality = info.get("card", "All")
113
114        proceed = False
115        selected_messages = {}
116
117        if next_act in self._object_params.get("create_relationship", {}):
118            type_objects = set(self._process.get_specific_type(obj_type).keys())
119            picked_messages = {
120                (item_id, action, ref)
121                for item_id, action, ref in self._process.board.messages
122                if action in trans_list
123            }
124
125            picked_items = {item_id for item_id, _, _ in picked_messages}
126            available = type_objects & picked_items
127            matched = type_objects & picked_items
128
129            if isinstance(cardinality, list):
130                min_val, max_val = map(int, cardinality)
131
132                if len(available) >= min_val:
133                    k = min(len(available), max_val)
134                    matched = set(random.sample(list(available), k))
135                    selected_messages = {
136                        (item_id, action, ref)
137                        for item_id, action, ref in picked_messages
138                        if item_id in matched
139                    }
140                    proceed = True
141                else: 
142                    return False, set()
143
144            elif isinstance(cardinality, int):
145                if len(available) >= cardinality:
146                    matched = set(random.sample(list(available), cardinality))
147                    selected_messages = {
148                        (item_id, action, ref)
149                        for item_id, action, ref in picked_messages
150                        if item_id in matched
151                    }
152                    proceed = True
153                else: 
154                    return False, set()
155
156            elif cardinality == "All":
157                proceed = matched == type_objects
158                selected_messages = picked_messages 
159
160            elif cardinality == "Any":
161                proceed = len(matched) >= 1
162                selected_messages = {
163                    (item_id, action, ref)
164                    for item_id, action, ref in picked_messages
165                    if item_id in matched
166                }
167                
168            elif cardinality == "CUSTOM":
169                proceed, matched = self._custom.custom_cardinality_rule(
170                    process=self._process, 
171                    current_object_id = self._id,
172                    available_objects=available,
173                )
174                selected_messages = {
175                    (item_id, action, ref)
176                    for item_id, action, ref in picked_messages
177                    if item_id in matched
178                }
179                
180            if proceed:
181                for obj in matched:
182                    self._process.set_relationships(self._id, obj)
183
184                self._process.board.remove_message(selected_messages)
185                return True, matched
186
187            return False, set()
188
189        else:
190            # objects currently related
191            id_target_object = {
192                x for x in self._process.get_relationships(self._id)
193                if x.split('_')[0] == obj_type
194            }
195
196            picked_messages = {
197                (item_id, action, ref)
198                for item_id, action, ref in self._process.board.messages
199                if action in trans_list and self._id in ref
200            }
201
202            picked_items = {item_id for item_id, _, _ in picked_messages}
203            matched = id_target_object & picked_items
204            
205            if id_target_object or picked_items:
206
207                if cardinality == "All":
208                    proceed = matched == id_target_object
209
210                elif cardinality == "Any":
211                    proceed = len(matched) >= 1
212
213                elif isinstance(cardinality, list):
214                    min_val, max_val = map(int, cardinality)
215                    proceed = min_val <= len(matched) <= max_val
216
217                elif isinstance(cardinality, int):
218                    proceed = len(matched) >= cardinality
219
220            if proceed:
221                self._process.board.remove_message(picked_messages)
222
223            return proceed, matched

396    def define_xor_next_activity(self, all_enabled_transition):
397        """ Three different methods to decide which path following from XOR gateway:
398        * Random choice: each path has equal probability to be chosen (AUTO)
399        ```json
400        "probability": {
401            "A_ACCEPTED": "AUTO",
402            "skip_2": "AUTO",
403            "A_FINALIZED": "AUTO",
404        }
405        ```
406        * Defined probability: in the file json it is possible to define for each path a specific probability (PROBABILITY as value)
407        ```json
408        "probability": {
409            "A_PREACCEPTED": 0.20,
410            "skip_1": 0.80
411        }
412        ```
413        * Custom method: it is possible to define a dedicate method that given the possible paths it returns the one to
414        follow, using whatever techniques the user prefers. (CUSTOM)
415        ```json
416        "probability": {
417            "A_CANCELLED": "CUSTOM",
418            "A_DECLINED": "CUSTOM",
419            "tauSplit_5": "CUSTOM"
420        }
421        ```
422        """
423        prob = ['AUTO'] if not self._object_params["probability"] else self._retrieve_check_paths(all_enabled_transition)
424        self._check_type_paths(prob)
425        if prob[0] == 'AUTO':
426            next = random.choices(list(range(0, len(all_enabled_transition), 1)))[0]
427        elif prob[0] == 'CUSTOM':
428            next = self.call_custom_xor_function(all_enabled_transition)
429        elif type(prob[0] == float()):
430        #elif isinstance(prob[0], float):
431            if self._check_probability(prob):
432                value = [*range(0, len(prob), 1)]
433                next = int(random.choices(value, prob)[0])
434            else:
435                next = random.choices(list(range(0, len(all_enabled_transition), 1)))[0]
436
437        return all_enabled_transition[next]

439    def define_processing_time(self, activity):
440        """ Three different methods are available to define the processing time for each activity:
441            * Distribution function: specify in the json file the distribution with the right parameters for each
442            activity, see the [numpy_distribution](https://numpy.org/doc/stable/reference/random/generator.html) distribution, (DISTRIBUTION).
443            **Be careful**: A negative value generated by the distribution is not valid for the simulator.
444            ```json
445             "processing_time": {
446                 "Add Item": { "name": "uniform", "parameters": { "low": 3600, "high": 7200}},
447             }
448            ```
449            * Custom method: it is possible to define a dedicated method that, given the activity and its
450            characteristics, returns the duration of processing time required. (CUSTOM)
451            ```json
452            "processing_time": {
453                 "Add Item":  { "name": "custom"}
454            }
455            ```
456            * Mixed: It is possible to define a distribution function for some activities and a dedicated method for the others.
457            ```json
458            "processing_time": {
459                 "Add Item":  { "name": "custom"},
460                 "Close Order":  { "name": "uniform", "parameters": { "low": 3600, "high": 7200}}
461            }
462            ```
463        """
464        try:
465            if self._object_params["processing_time"][activity]["name"] == 'custom':
466                duration = self.call_custom_processing_time()
467            else:
468                distribution = self._object_params["processing_time"][activity]['name']
469                parameters = self._object_params["processing_time"][activity]['parameters']
470                duration = getattr(np.random, distribution)(**parameters, size=1)[0]
471                if duration < 0:
472                    print(f"WARNING: Negative processing time {duration} of activity {activity}")
473                    duration = 0
474        except:
475            raise ValueError("ERROR: The processing time of", activity, "is not defined in json file")
476        return duration

152                    }

165                    for item_id, action, ref in picked_messages
